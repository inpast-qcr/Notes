<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>11.div,span,p</title>
</head>
<body>
    <!--对文本进行封装
        div  标签默认独占1行
        span 标签长度是封装数据的长度
        p    标签是段落标签，默认在段落的上方和下方各空出1行（如果已有则不空）
    -->

    <div>div标签1</div>
    <div>div标签2</div>
    <div>div标签3</div>
    <span>span标签1</span>
    <span>span标签2</span>
    <span>span标签3</span>
    <p>p段落标签1</p>
    <p>p段落标签2</p>
    <p>p段落标签3</p>
</body>
</html>
